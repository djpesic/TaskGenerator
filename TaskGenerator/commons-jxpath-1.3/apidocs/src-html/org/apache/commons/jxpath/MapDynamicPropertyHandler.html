017    package org.apache.commons.jxpath;
018    
019    import java.util.Iterator;
020    import java.util.Map;
021    import java.util.Set;
022    
023    /**
024     * Implements the DynamicPropertyHandler interface for {@link java.util.Map}.
025     *
026     * @author Dmitri Plotnikov
027     * @version $Revision: 652845 $ $Date: 2008-05-02 12:46:46 -0500 (Fri, 02 May 2008) $
028     */
029    public class MapDynamicPropertyHandler implements DynamicPropertyHandler {
030    
031        public String[] getPropertyNames(Object object) {
032            Map map = (Map) object;
033            Set set = map.keySet();
034            String[] names = new String[set.size()];
035            Iterator it = set.iterator();
036            for (int i = 0; i < names.length; i++) {
037                names[i] = String.valueOf(it.next());
038            }
039            return names;
040        }
041    
042        public Object getProperty(Object object, String propertyName) {
043            return ((Map) object).get(propertyName);
044        }
045    
046        public void setProperty(Object object, String propertyName, Object value) {
047            ((Map) object).put(propertyName, value);
048        }
049    }
